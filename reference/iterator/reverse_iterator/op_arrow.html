operator->

pointer operator->() const;

概要

イテレータを通してオブジェクトにアクセスする

戻り値

&(operator*())

例

#include <iostream>
#include <vector>
#include <iterator>

class value {
  int x;

public:
  value(int x = 0) : x(x) {}
  int get() const { return x; }
};

int main()
{
  std::vector<value> v = {1, 2, 3};

  std::reverse_iterator<decltype(v)::iterator> it(v.end());

  int result = it->get();

  std::cout << result << std::endl;
}

出力

3
