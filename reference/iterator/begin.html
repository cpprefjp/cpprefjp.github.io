begin (C++11)

namespace std {
  template <class C>
  auto begin(C& c) -> decltype(c.begin());       // (1)

  template <class C>
  auto begin(const C& c) -> decltype(c.begin()); // (2)

  template <class T, size_t N>
  T* begin(T (&array)[N]);                       // (3) C++11

  template <class T, size_t N>
  constexpr T* begin(T (&array)[N]) noexcept;    // (3) C++14
}

概要

範囲から先頭要素へのイテレータを取得する。

戻り値

  • (1) : return c.begin();
  • (2) : return c.begin();
  • (3) : return array;
