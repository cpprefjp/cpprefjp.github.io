operator[] (C++11)

unspecified operator[](difference_type n) const;

概要

任意の位置にランダムアクセスする。

戻り値

return std::move(base()[n]);

例

#include <iostream>
#include <vector>
#include <memory>
#include <iterator>

int main()
{
  std::vector<std::unique_ptr<int>> v;
  for (int i = 0; i < 5; ++i)
    v.emplace_back(new int(i));

  auto it = std::make_move_iterator(v.begin());
  std::unique_ptr<int> p = it[2];

  std::cout << *p << std::endl;
}

2
