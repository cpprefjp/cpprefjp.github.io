概要

ナノ秒単位を表現するdurationの別名

例

#include <iostream>
#include <chrono>

int main()
{
  std::chrono::nanoseconds ns1(30);
  std::chrono::nanoseconds ns2(20);

  std::chrono::nanoseconds result = ns1 + ns2;

  std::cout << result.count() << std::endl;
}
